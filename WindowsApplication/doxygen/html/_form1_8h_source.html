482 private: System::Void backgroundWorker1_DoWork(System::Object^ sender, System::ComponentModel::DoWorkEventArgs^ e) {
483  double uSteering; // controller output steering
484  double uThrottle; // controller output throttle
485  int steering=15; // discrete steering shifted
486  int oldSteering= 128 + 15; // previous loop discrete steering
487  int throttle=30; // discrete throttle shifted
488  int oldThrottle=30; // privious loop discrete throttle
489  double radius; // distance between the origin and the car
490  double velocity; // velocity of the car
491  double Xold; // X-coordinate of the previous loop
492  double Yold; // Y-coordinate of the previous loop
493  double dt; // elapsed time since last frame [sec.]
494  double time = 0; // total elapsed time
495  LARGE_INTEGER freq; // ticks per second, for accurate time measurement
496  LARGE_INTEGER t1, t2, t3; // ticks, for accurate time measurement
497  double vD, dD; // direction of velocity vector, angel difference
498 
499  // Create txt-File
500  DataTxt.std::ofstream::open("Data.txt");
501  DataTxt << "time\tX1\tY1\tRZ1\tv\n" ;
502 
503  QueryPerformanceFrequency(&freq); // get ticks per second
504  QueryPerformanceCounter(&t1); // get starting time
505  QueryPerformanceCounter(&t3);
506 
507  Serial *mycomport = new Serial("COM3");
508  PID_Controller controller1(steeringMin, steeringMax);
509  PID_Controller controller2(throttleMin, throttleMax);
510 
511  //set old coordinates to current coordinates, this prevents an instantanious jump of the car in the first loop
512  getCoordinates("Car_1", X, Y, D);
513  Xold = X;
514  Yold = Y;
515 
516  while (true){
517  //check for cancellation
518  if (this->backgroundWorker1->CancellationPending)
519  break;
520 
521  //get current time
522  QueryPerformanceCounter(&t2);
523  time = ((double)(t2.QuadPart - t1.QuadPart)) / freq.QuadPart;
524 
525  // elapsed time since last frame [sec.]
526  dt = ( (double)(t2.QuadPart - t3.QuadPart) ) / freq.QuadPart;
527 
528  //get coordinates (waits for new frame (Vicon Tracker frame rate) )
529  getCoordinates("Car_1", X, Y, D);
530 
531  radius = sqrt(X*X+Y*Y); // [mm]
532  velocity = sqrt((X-Xold)*(X-Xold)+(Y-Yold)*(Y-Yold)) / dt; // [mm/sec]
533 
534  //detect direction of velocity vector
535  vD = atan2(Y-Yold,X-Xold); //vD = [-pi;pi]
536  dD = abs(D - vD); //angle difference
537  if ((dD > (pi/2)) && (dD < (3/2*pi))) // car moves backward
538  velocity = -velocity;
539 
540  //PID-controllers
541  uSteering = controller1.step(wRadius , radius, KPs, Tns, Tvs, dt);
542  uThrottle = controller2.step(wVelocity, velocity, KPt, Tnt, Tvt, dt);
543 
544  //send data to car
545  steering = (int) uSteering + 15;
546  steering |= 1 << 7;
547  if (steering != oldSteering)
548  mycomport->WriteData(steering, 1); //steering
549  oldSteering = steering;
550 
551  throttle = (int) -uThrottle + 30;
552  if (throttle != oldThrottle)
553  mycomport->WriteData(throttle, 1); //throttle
554  oldThrottle = throttle;
555 
556  //write txt
557  DataTxt << time << "\t" //time
558  << X << "\t" << Y << "\t" << D << "\t"
559  << velocity << "\t" << (int) uThrottle << "\n";
560 
561  //prepare next loop
562  t3 = t2;
563  Xold = X;
564  Yold = Y;
565 
566  //redraw pictureBox1
567  pictureBox1->Invalidate();
568  }
569  mycomport->WriteData( 30, 1); //Stop motor
570  mycomport->WriteData(128 +15, 1); //steering angle = 0
571  delete mycomport;
572  }
